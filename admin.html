Firestore Rules ঠিক করুন
Firestore permission-denied error — Firebase Rules এ Delete এর অনুমতি নেই। নিচের Rules টি কপি করে Firebase Console এ লাগালেই ঠিক হয়ে যাবে।
📋 ধাপ ১ — Firebase Console খুলুন
Firebase Console Firestore Rules খুলুন
📋 ধাপ ২ — এই Rules কপি করে Replace করুন
rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {

    function isAdmin() {
      return request.auth != null && (
        request.auth.token.email == '[email protected]' ||
        request.auth.uid == 'E97OGEv5ZJFbHaoojCnFe25GSeY1' ||
        exists(/databases/$(database)/documents/autoscrip_admins/$(request.auth.uid))
      );
    }

    match /autoscrip_tools/{docId} {
      allow read: if true;
      allow write, delete: if isAdmin();
    }
    match /autoscrip_users/{docId} {
      allow read: if isAdmin();
      allow write: if request.auth != null &&
        (request.auth.uid == docId || isAdmin());
    }
    match /autoscrip_admins/{docId} {
      allow read, write: if isAdmin() || request.auth.uid == docId;
    }
    match /autoscrip_requests/{docId} {
      allow read, write, delete: if isAdmin();
      allow create: if request.auth != null;
    }
    match /autoscrip_notifications/{docId} {
      allow read: if request.auth != null;
      allow write, delete: if isAdmin();
    }
    match /autoscrip_subscribers/{docId} {
      allow read, write: if isAdmin() ||
        (request.auth != null && request.auth.uid == docId);
    }
    match /autoscrip_youtube/{docId} {
      allow read: if true;
      allow write, delete: if isAdmin();
    }
    match /autoscrip_comments/{docId} {
      allow read: if true;
      allow create: if request.auth != null;
      allow update, delete: if isAdmin();
    }
    match /autoscrip_config/{docId} {
      allow read: if true;
      allow write: if isAdmin();
    }
    match /autoscrip_activity/{docId} {
      allow read, write: if isAdmin();
    }
    match /{collection}/{docId} {
      allow read, write, delete: if isAdmin();
    }
  }
}
Rules paste করে Firebase Console এ Publish বাটন চাপুন। তারপর এই পেজ রিফ্রেশ করুন — ডিলিট কাজ করবে।
